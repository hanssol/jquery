		vp1 = document.getElementsByTagName('p')[0];
		vp1.style.background = 'pink';
		
		dom객체요소가 css jQeury메소드를 사용할 수 없다.
		vp1.css('font-size','2.0em');

		dom객체요소에 $()로 감싸주면 jQuery객체로 변환되어
		css jQuery메소드를 사용할 수 있다.
		$(vp1).css('font-size','2.0em');
